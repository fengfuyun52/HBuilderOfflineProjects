switch组件

基本参数说明


this.num = new UISelect({
  //数据模型，会影响插件渲染
  datamodel: {
    checkedFlag: false,//当前选中状态
    checkedClass: 'current'//选中状态时dom的class
  },
  //选项改变执行的事件
  changed: function (status) {
    console.log(status);
  };
  //UI组件包裹器，会放到什么容器里面
  wrapper: this.$('.simple_switch')
});

switch简单应用

当前选择状态：

this.uiSwitch = new UISwitch({
  wrapper: this.$el.find('.simple_switch'),
  changed: function (status) {
    scope.$('.demo1Status').html(status);
  }
});
this.uiSwitch.show();

this.$('.demo1Status').html(this.uiSwitch.getStatus());

设置默认值


this.uiSwitch1 = new UISwitch({
  datamodel: {
    checkedFlag: true
  },
  wrapper: this.$el.find('.simple_switch1')
});
this.uiSwitch1.show();
