基本参数说明


    @param template     {String}    自定义模板

注意项


/*
① 如果不是使用showToast的话，在view离开时（onhide）时候需要做组件的销毁或者隐藏
② 组件本身继承至layer类，所以具有以下特性
//需要蒙版
this.needMask = true;

//需要点击蒙版删除
this.maskToHide = true;

//需要居中定位
this.needReposition = true;

//是否具有后退关闭弹出层需求
this.hasPushState = false;;

//类型为layer
this.type = 'layer';
*/

框架用法


/*
框架提供一个简单的提示类效果，并且此弹出框是单例的全局共享，在后退等操作中会自动关闭，建议使用

title为显示的内容
timeout为多少秒关闭toast
hideAction为关闭时执行的回调
clickToHide，布尔型是否需要点击mask关闭组件
showToast: function (title, timeout, hideAction, clickToHide) {
*/
//只传入一个参数，即为toast内容
this.showToast({
    datamodel: {
       content: '框架使用'
    }
});
//显示toast，不带蒙层
this.showToast({
    datamodel: {
       content: '框架使用-不带蒙版'
    },
    needMask: false
});
//设置自动关闭时间
this.showToast({
    datamodel: {
       content: '带蒙版，点击蒙版不关闭，5秒关闭'
    },
    maskToHide: false,
    hideSec: 5000
});
//重写animateShowAction，控制动画
this.showToast({
    datamodel: {
       content: '动画'
    },
    animateShowAction: function (el) {
       el.css({
          '-webkit-transform': 'translate(0, -50%)',
          transform: 'translate(0,  -50%)'
       });
       el.show().animate({
          '-webkit-transform': 'translate(0, 0)',
           transform: 'translate(0, 0)'
       }, 200, 'ease-in-out', $.proxy(function () {
          this.$el.css({
             '-webkit-transform': '',
              transform: ''
          });
       }, this));
       }
    });
