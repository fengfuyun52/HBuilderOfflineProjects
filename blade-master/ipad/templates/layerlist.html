var scope = this;
if (!this.list) {
  var data = [];

  for (var i = 0; i < 4; i++) {
    data.push({ name: '更多操作' + i });
  }

  this.list = new UILayerList({
    datamodel: {
      list: data
    },
    onItemAction: function (item) {
      scope.$('.listSec').html(item.name);
      this.hide();
    }
  });
}

this.list.show();
